create database BEP1;
use BEP1;
create table user(
	user_id int primary key,
    usrename char(10) not null,
    phoneNumber char(20) not null,
    address char(50) not null,
    sex char(5) not null
);
create table flow(
	flow_id int primary key,
	targetName char(10) not null,
    count numeric(10,1),
    ReVisitedRate numeric(10,1),
    stamp int
);
create table wlan(
	wlan_id int primary key,
    totalRecords int,
    timestamp int,
    unit char(10) not null,
    uplinkRate int,
    downlinkRate int
);
create table deviceFlow(
	device_id char(60) primary key,
	deviceName char(50) not null,
    upTraffic int,
    downTraffic int,
    unit char(50) not null
);
create table deviceDetail(
	detail_id char(60) primary key,
	name char(50) not null,
    esn char(50) not null,
    deviceIp char(50) not null,
    neType char(50) not null,
    status int,
    gisLon numeric(10,1),
    gisLat numeric(10,1),
    traffic numeric(10,1),
    onlineUsers int,
    cpuRate int,
    upwardSpeed numeric(10,1),
    downwardSpeed numeric(10,1),
    mac char(50) not null
);
create table siteHealth(
	site_id char(60) primary key,
    deviceHealth int,
    radioHealth int,
    siteHealth int
);
create table userNumber(
	number_id char(60) primary key,
    exprWorse int,
    exprBad int,
    exprGood int,
    exprBetter int,
    oneHour int,
    oneToTwoHour int,
    twoToFourHour int,
    fourToSixHour int,
    sixToEightHour int,
    eightHour int
);
